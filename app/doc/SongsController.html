def create
  @song = @user.songs.new song_params
  if @song.save
    # Record song's preoprties to predictor
    recommender = SongRecommender.new
    recommender.add_to_matrix!(:topics, @song.genre_tag, @song.id)

    redirect_to user_path(@user)
  else
    redirect_to user_path(@user)
  end
end
